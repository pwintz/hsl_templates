
\begin{frame}[t]{Modeling Framework: Hybrid Dynamical Systems}
  \vspace{-12pt}

  % We consider \emph{hybrid dynamical systems}, which are written as
  \begin{equation*}
      \calH: \left\{\begin{aligned}
          \mstructure{\dot x \hspace{0.5em}}&\mstructure{{}= f(x)} & \mstructure{x \in C} \\
          \darkred{x^+} &= \darkred{g(x)} & \darkred{x \in D}
      \end{aligned}\right. 
      \label{eq:hybrid system}
  \end{equation*}%\pause
  % with 
  \begin{multicols}{2}
  \begin{itemize}
    % \renewcommand{\emph}[1]{#1}
    \item<2-> \structure{flow set $C \subset \realsn$} %\pause
    \item<4-> \structure{flow map $f : {C} \to \realsn$} %\pause
    \item<3->[\darkred{$\blacktriangleright$}] \darkred{jump set $D \subset \realsn$} %\pause
    \item<5->[\darkred{$\blacktriangleright$}] \darkred{jump map $g : {D} \to \realsn$} %\pause
  \end{itemize}
\end{multicols}
